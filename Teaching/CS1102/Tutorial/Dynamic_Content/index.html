Part 2

  • Task 2.1: First and Last Name Formats
    Open HTML File 
    <!DOCTYPE html>
<html>
<head>
    <title>Task 2.1</title>
    <script>
        function greeting() {
            var firstname, lastname;
            firstname = prompt("What is your first name?");
            lastname = prompt("What is your last name?");
            
            var output = "Hello " + lastname + " " + firstname + "!<br/><br/>";
            output += "Different ways to display your name:<br/>";
            output += firstname + " " + lastname + "<br/>";
            output += lastname + ", " + firstname;
            
            document.getElementById("message").innerHTML = output;
        }
    </script>
</head>
<body onload="greeting();">
    <div id="message"></div>
</body>
</html>
  • Task 2.2: Update Other HTML Elements (Fruit List)
    Open HTML File 
    <!DOCTYPE html>
<html>
<head>
    <title>Task 2.2</title>
    <script>
        function addFruit() {
            var fruit, s;
            s = document.getElementById("fruit").innerHTML;
            fruit = prompt("Enter a fruit that you like");
            
            // Checks if user actually entered something
            if (fruit !== null && fruit !== "") {
                document.getElementById("fruit").innerHTML = "<li>" + fruit + "</li>" + s;
            }
        }
        
        function clearFruits() {
            document.getElementById("fruit").innerHTML = "";
        }
    </script>
</head>
<body>
    <h2>
        My favorite fruits
        <span style="cursor:pointer; color:blue;" onclick="addFruit();">[Add]</span>
        <span style="cursor:pointer; color:red;" onclick="clearFruits();">[Clear]</span>
    </h2>
    <ul id="fruit"></ul>
</body>
</html>
